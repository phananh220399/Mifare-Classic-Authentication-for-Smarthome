unsigned long Get_Epoch_Time() {

void dump_byte_array(byte *buffer, byte bufferSize) {
  for (byte i = 0; i < bufferSize; i++) {
    Serial.print(buffer[i] < 0x10 ? " 0" : " ");
    Serial.print(buffer[i], HEX);
  }
}

bool check_new_nfc_card() {
  // In this sample we use the second sector,
  // that is: sector #1, covering block #4 up to and including block #7
  //byte sector = 1;
  //byte blockAddr = 4;

  byte trailerBlock = 43;
  MFRC522::StatusCode status;
  byte buffer[18];
  byte size = sizeof(buffer);

  MFRC522::MIFARE_Key key_nfc;
  for (byte i = 0; i < 6; i++) {
    key_nfc.keyByte[i] = 0xFF;
  }

  Serial.println();
  Serial.println("check_new_nfc_card");

  Serial.print(F("authen with key: "));
  dump_byte_array(key_nfc.keyByte, 6);
  Serial.println();

  // Authenticate using key A
  status = (MFRC522::StatusCode)mfrc522.PCD_Authenticate(MFRC522::PICC_CMD_MF_AUTH_KEY_B, trailerBlock, &key_nfc, &(mfrc522.uid));
  Serial.println(status);
  if (status != MFRC522::STATUS_OK) {
    Serial.print(F("PCD_Authenticate() failed: "));
    Serial.println(mfrc522.GetStatusCodeName(status));
    return false;
  }

  return true;
}

bool change_key_B() {
  byte blockAddr = 43;
  MFRC522::StatusCode status;
  byte buffer[18];
  byte size = sizeof(buffer);

  Serial.println();
  Serial.println("change_key_B");

  byte data_write[16];

  // key A
  data_write[0] = 0xFF;
  data_write[1] = 0xFF;
  data_write[2] = 0xFF;
  data_write[3] = 0xFF;
  data_write[4] = 0xFF;
  data_write[5] = 0xFF;

  // access bit
  data_write[6] = 0xFF;
  data_write[7] = 0x07;
  data_write[8] = 0x80;
  data_write[9] = 0x69;

  // key B
  for (int i = 0; i < 6; i++) {
    data_write[i + 10] = key_authen.keyByte[i];
  }

  Serial.print(F("data_write: "));
  dump_byte_array(data_write, 16);
  Serial.println();
  status = (MFRC522::StatusCode)mfrc522.MIFARE_Write(blockAddr, data_write, 16);
  if (status != MFRC522::STATUS_OK) {
    Serial.print(F("MIFARE_Write() failed: "));
    Serial.println(mfrc522.GetStatusCodeName(status));
    return false;
  }
  else{
    buzzerSuccess();
  }
  // Serial.println();

  return true;
}

bool rst_key_to_default() {
  byte blockAddr = 43;
  MFRC522::StatusCode status;
  byte buffer[18];
  byte size = sizeof(buffer);

  Serial.println();
  Serial.println("rst_key_to_default");

  // Serial.println(F("Authenticating using key B..."));
  status = (MFRC522::StatusCode)mfrc522.PCD_Authenticate(MFRC522::PICC_CMD_MF_AUTH_KEY_B, blockAddr, &key_authen, &(mfrc522.uid));
  Serial.println(status);
  if (status != MFRC522::STATUS_OK) {
    Serial.print(F("PCD_Authenticate() failed: "));
    Serial.println(mfrc522.GetStatusCodeName(status));
    return false;
  }

  byte data_write[16];

  // key A
  data_write[0] = 0x00;
  data_write[1] = 0x00;
  data_write[2] = 0x00;
  data_write[3] = 0x00;
  data_write[4] = 0x00;
  data_write[5] = 0x00;

  // access bit
  data_write[6] = 0xFF;
  data_write[7] = 0x07;
  data_write[8] = 0x80;
  data_write[9] = 0x69;

  // key B
  for (int i = 0; i < 6; i++) {
    data_write[i + 10] = 0xFF;
  }

  Serial.print(F("data_write: "));
  dump_byte_array(data_write, 16);
  Serial.println();
  status = (MFRC522::StatusCode)mfrc522.MIFARE_Write(blockAddr, data_write, 16);

  if (status != MFRC522::STATUS_OK) {
    Serial.print(F("MIFARE_Write() failed: "));
    Serial.println(mfrc522.GetStatusCodeName(status));
    return false;
  } else {
    buzzerSuccess();
  }
  Serial.println("reset nfc card finished.");
  return true;
}

int check() {
  if (ss == "5E AE 04 7B" || ss == "B3 DF E7 F4") {
    return 1;
  } else if (ss == "C3 E8 56 FC" || ss == "B3 FD 00 98") {
    return 2;
  } else if (ss == "33 C3 58 95" || ss == "B3 C2 6F FB") {
    return 3;
  } else return 0;
}

void buzzerSuccess() {
  tone(BUZZER_PIN, 2300);
