void RFID() {
  position = 1;
  if (!mfrc522.PICC_IsNewCardPresent())
    return;

  // Select one of the cards
  if (!mfrc522.PICC_ReadCardSerial())
    return;

  if (last_comm_failed) {
    if (mfrc522.uid.uidByte[0] == last_uid[0] && mfrc522.uid.uidByte[1] == last_uid[1] && mfrc522.uid.uidByte[2] == last_uid[2] && mfrc522.uid.uidByte[3] == last_uid[3] && (millis() - u32_last_cmm_mil < 3000)) {
      // same uid
      return;
    }
  }

  for (int i = 0; i < mfrc522.uid.size; i++) {
    last_uid[i] = mfrc522.uid.uidByte[i];
  }

  // Show some details of the PICC (that is: the tag/card)
  Serial.print(F("Card UID:"));
  dump_byte_array(mfrc522.uid.uidByte, mfrc522.uid.size);
  //Serial.println(mfrc522.uid.size);
  Serial.println();
  Serial.print(F("PICC type: "));
  MFRC522::PICC_Type piccType = mfrc522.PICC_GetType(mfrc522.uid.sak);
  Serial.println(mfrc522.PICC_GetTypeName(piccType));

  // Check for compatibility
  if (piccType != MFRC522::PICC_TYPE_MIFARE_MINI && piccType != MFRC522::PICC_TYPE_MIFARE_1K && piccType != MFRC522::PICC_TYPE_MIFARE_4K) {
    Serial.println(F("This sample only works with MIFARE Classic cards."));
    u32_last_cmm_mil = millis();
    return;
  }
  if (check_uid()) {

    key_authen.keyByte[0] = 0x22;
    key_authen.keyByte[1] = 0x23;
    key_authen.keyByte[2] = 0x24;
    key_authen.keyByte[3] = 0x25;
    key_authen.keyByte[4] = 0x26;
    key_authen.keyByte[5] = 0x27;

    if (check_card_by_system_key()) {

      if (check_pass() == 1) {
        buzzerSuccess();
        Serial.println("Card on system, openning door room 1.....");
        myservo.write(0);
        delay(2000);
        myservo.write(90);

        active_open_door = true;
        last_comm_failed = false;
      }
    }
  }

  else if (!check_uid()) {

    key_authen.keyByte[0] = 0x32;
    key_authen.keyByte[1] = 0x33;
    key_authen.keyByte[2] = 0x34;
    key_authen.keyByte[3] = 0x35;
    key_authen.keyByte[4] = 0x36;
    key_authen.keyByte[5] = 0x37;

    if (check_card_by_system_key()) {

      if (check_pass() == 0) {
        buzzerSuccess();
        Serial.println("Card on system, openning door room 1.....");
        myservo.write(0);
        delay(2000);
        myservo.write(90);

        active_open_door = true;
        last_comm_failed = false;
      }
    }
  }

  else {
    last_comm_failed = true;
  }

  u32_last_cmm_mil = millis();

  // Halt PICC
  mfrc522.PICC_HaltA();
  // Stop encryption on PCD
  mfrc522.PCD_StopCrypto1();
}

bool check_uid() {
  if (ss == "5E AE 04 7B" || ss == "B3 DF E7 F4") return true;
  else if (ss == "C3 E8 56 FC" || ss == "B3 FD 00 98") return true;
  else if (ss == "33 C3 58 95" || ss == "B3 C2 6F FB") return true;
  else return false;
}

void buzzerSuccess() {
  tone(BUZZER_PIN, 2300);
  delay(100);
  noTone(BUZZER_PIN);
  delay(100);
  tone(BUZZER_PIN, 2300);

}

}
